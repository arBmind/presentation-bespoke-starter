class NestedStack
  def initialize tree
    @stack = [tree]
  end
  def push
    nested = []
    @stack[-1] << nested
    @stack << nested
  end
  def pop
    @stack.pop
  end
  def pass l
    @stack[-1] << l
  end
end
def nested_array array, &block
  tree = []
  api = NestedStack.new tree
  array.each do |l|
    block.call api, l
  end
  tree
end

lines = ["a", ">", "b", ">", "c", "<", "<", "d"]
nested_array(lines) do |a, l|
  if l == ">"
    a.push
  elsif l == "<"
    a.pop
  else
    a.pass l
  end
end
----
